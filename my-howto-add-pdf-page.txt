Subject: Question on a PDF file's resources

my $xobject =  $pdfo.page(1)<Resources><XObject>{$key};

See PDF-API6

Page Methods
  to-xobject

Example:
  ...
  # open desired page from parent pdf


  # Import first page from the old PDF
  # Add it to the new PDF's first page...

  $gfx.do($xo);

=begin code
# my attempt
use PDF::API6;
use PDF::Page;
use PDF::XObject;
use PDF::Content;
my PDF::API6 $old .= open('our/old.pdf');
my PDF::API6 $pdf .= new;
my PDF::Page $page = $pdf.add-page;
my PDF::Content $gfx = $page.gfx;

# Import first page from the old PDF
my PDF::XObject $xo = $old.page(1).to-xobject;

# Add it to the new PDF's first page at 1/2 scale
my $width = $xo.width / 2;
my $bottom = 5;
my $left = 10;
$gfx.do($xo, :position[$bottom, $left], :$width);

$pdf.save-as('our/new.pdf');

=end code
